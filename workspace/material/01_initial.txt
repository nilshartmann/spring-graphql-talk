type Rating {
  id: ID!

  comment: String!
  stars: Int!
}

type Beer  {
    id: ID!
    name: String!
    """
    Price must not be higher than **ten euros**
    """
    price: String!
    ratings: [Rating!]!
}

type User {
  id: ID!
  login: String!
  name: String!
}

extend type Rating {
  author: User!
}




type Query {
    beer(beerId: ID): Beer
    beers: [Beer!]!
}
